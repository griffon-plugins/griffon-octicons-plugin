233     private static final String ERROR_DESCRIPTION_BLANK = "Argument 'description' must not be blank";
234 
235     public static OctIconFont findByDescription(@Nonnull String description) {
236         requireNonBlank(description, "Icon description must not be blank.");
237 
238         String[] parts = description.split(":");
239         for (OctIconFont font : values()) {
240             if (font.getDescription().equals(parts[0])) {
241                 return font;
242             }
243         }
244 
245         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
246     }
247 
248     private String description;
249     private char code;
250 
251     OctIconFont(@Nonnull String description, char code) {
252         this.description = description;
253         this.code = code;
254     }
255 
256     @Nonnull
257     public String getDescription() {
258         return description;
259     }
260 
261     public char getCode() {
262         return code;
263     }
264 
265     @Nonnull
266     public static IllegalArgumentException invalidDescription(@Nonnull String description) {
267         requireNonBlank(description, ERROR_DESCRIPTION_BLANK);
268         throw new IllegalArgumentException("Description " + description + " is not a valid OctIcon icon description");
269     }
270 
271     @Nonnull
272     public static IllegalArgumentException invalidDescription(@Nonnull String description, Exception e) {
273         requireNonBlank(description, ERROR_DESCRIPTION_BLANK);
274         throw new IllegalArgumentException("Description " + description + " is not a valid OctIcon icon description", e);
275     }
